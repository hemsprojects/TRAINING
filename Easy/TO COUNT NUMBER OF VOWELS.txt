
class Main {
    public static void main(String[] args) {
        String b="HemAnath";
        String s=b.toLowerCase();
        int sum=0;
        for(int i=0;i<s.length();i++)
        {
            char c=s.charAt(i);
            if(c=='a'||c=='e'||c=='i'||c=='o'||c=='u')
            {
                sum+=1;
            }
        }
        System.out.println(sum);
    }
}
